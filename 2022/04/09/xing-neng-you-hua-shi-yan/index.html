性能优化实验

以下代码只供参考，至于怎样展开，双重循环展开怎么搭配效果会更好，还是需要不断试验。

实验一 poly.c的优化

代码：

1、常系数 const_poly_eval：不知道该怎么优化，感觉老师给的代码就很不错，直接套用。

int const_poly_eval(int *not_use, int not_use2, int x)
{
	//int y = x * x;
	//int z = y*x;
    //return 81 + (((y - z) + ((x << 6) - (x << 1))) + (((y << 6) + (y << 1)) + ((z << 6) + (z << 4))));
	int x64, x16, x2;
	x64 = x << 6;
	x16 = x << 4;
	x2 = x << 1;
	return 81 + x64 - x2 + (x64 + x2 + x + (x64 + x16 - x) * x) * x;
	
}

2、任意的高阶（和10阶） my_poly_eval：主要考虑循环展开，经过试验，当展开>=10时，几乎性能就达到最优了。同时需要单独考虑degree=10的情况，为了让计算十阶更快一点，直接全部展开，并减少“昂贵”的乘法的使用。

int my_poly_eval(int *a, int degree, int x)
{
	if (degree == 10) {
		return a[0] + (a[1] + (a[2] + (a[3]+ (a[4]+ (a[5]+ (a[6]+ (a[7]+ (a[8]+ (a[9]+ a[10] * x)*x)*x)*x)*x)*x)*x)*x)*x)*x;
		//return a[0] + a[1] * x + a[2] * xpwr2 + a[3] * xpwr3 + a[4] * xpwr4 + a[5] * xpwr5 + a[6] * xpwr6 + a[7] * xpwr7 + a[8] * xpwr8 + a[9] * xpwr9 + a[10] * xpwr10;
	}
	else {
		int i;
		//int x2 = x * x;
		//int x5 = x2 * x2 * x;
		//int x11 = x5 * x5 * x;
		int result = 0;
		int xpwr2 = x * x;
		int xpwr3 = xpwr2 * x;
		int xpwr4 = xpwr3 * x;
		int xpwr5 = xpwr4 * x;
		int xpwr6 = xpwr5 * x;
		int xpwr7 = xpwr6 * x;
		int xpwr8 = xpwr7 * x;
		int xpwr9 = xpwr8 * x;
		int xpwr10 = xpwr9 * x;
		int x11 = xpwr10 * x;
		int limit11 = degree - 11;
		int limit5 = degree - 5;
		int cnt = 1;
		for (i = 0; i <= limit11; i += 11) {
			result += (a[i] + a[i + 1]*x + a[i + 2] * xpwr2 + a[i + 3] * xpwr3 + a[i + 4] * xpwr4 + a[i + 5] * xpwr5 + a[i + 6] * xpwr6 + a[i + 7] * xpwr7 + a[i + 8] * xpwr8 + a[i + 9] * xpwr9 + a[i + 10] * xpwr10) * cnt;
			//result += (a[i] + (a[i + 1] + (a[i + 2] + (a[i + 3] + (a[i + 4] + (a[i + 5] + (a[i + 6] + (a[i + 7] + (a[i + 8] + (a[i + 9] + a[i + 10] * x) * x) * x) * x) * x) * x) * x) * x) * x) * x) * cnt;
			cnt *= x11;
		}
		for (;i <= limit5;i += 5) {
			result += (a[i] + a[i + 1] * x + a[i + 2] * xpwr2 + a[i + 3] * xpwr3 + a[i + 4] * xpwr4) * cnt;
			//result += (a[i] + (a[i + 1] + (a[i + 2] + (a[i + 3] + a[i + 4] * x)*x)*x)*x) * cnt;
			cnt *= xpwr5;
			//cnt *= x5;
		}
		for (;i <= degree;i++) {
			result += a[i] * cnt;
			cnt *= x;
		}
		return result;
	}
}

{
	int i, j, limitj, limiti;
	for (i = 0;i < N;i++) {

	}

	limiti = N - 7;

		for (j = 0; j <= limitj; j += 11) {
			colsum[j] += M[i][j] + M[i + 1][j] + M[i + 2][j] + M[i + 3][j] + M[i + 4][j] + M[i + 5][j] + M[i + 6][j];// + M[i + 7][j] + M[i + 8][j] + M[i + 9][j] + M[i + 10][j];

			colsum[j + 7] += M[i][j + 7] + M[i + 1][j + 7] + M[i + 2][j + 7] + M[i + 3][j + 7] + M[i + 4][j + 7] + M[i + 5][j + 7] + M[i + 6][j + 7];// + M[i + 7][j + 7] + M[i + 8][j + 7] + M[i + 9][j + 7] + M[i + 10][j + 7];

			colsum[j + 10] += M[i][j + 10] + M[i + 1][j + 10] + M[i + 2][j + 10] + M[i + 3][j + 10] + M[i + 4][j + 10] + M[i + 5][j + 10] + M[i + 6][j + 10];// + M[i + 7][j + 10] + M[i + 8][j + 10] + M[i + 9][j + 10] + M[i + 10][j + 10];
		}
		for (;j < N;j++) {
			colsum[j] += M[i][j] + M[i + 1][j] + M[i + 2][j] + M[i + 3][j] + M[i + 4][j] + M[i + 5][j] + M[i + 6][j];// + M[i + 7][j] + M[i + 8][j] + M[i + 9][j] + M[i + 10][j];

	for (;i < N;i++) {

			colsum[j + 3] += M[i][j + 3];

			colsum[j + 8] += M[i][j + 8];
			colsum[j + 9] += M[i][j + 9];
			colsum[j + 10] += M[i][j + 10];
		}
		for (;j < N;j++) {
			colsum[j] += M[i][j];
		}
	}
}
